def EMA(x): ## calculate the exponential moving average
    y=[]
    z=[]
    for i in range (x):
            y.append(random.randint(5000, 6000)/100)
            i = i+1
    lambda_EMA = 2/(x+1)

    
    #z.append((sum(y)/len(y))*(1-lambda_EMA)+y[0]*lambda_EMA )  
    
    return (sum(y)/len(y))*(1-lambda_EMA)+y[0]*lambda_EMA

def WMA(x): ## calcualte using weighted moving average 
    y=[]
    z=[]
    for i in range(x):      
        y.append(random.randint(5000, 6000)/100)
        z.append(y[i]*(x-i))
        i=i+1 
    return sum(z)/((x*(x+1))/2)
